  <script>
    // ** อย่าลืมนำ URL ของ Web App อันเดิมมาใส่ตรงนี้นะครับ **
    const GAS_URL = "https://script.google.com/macros/s/AKfycbzbQIdnzCmFSn9ENUqXqHTRYsH4Q_XQxSAzW6Sf5krNH0VCZLt1Eyff5Jiqm1l5EZ9F3w/exec";
    let appData = { students: [], activities: [], schedule: [] };

    window.onload = async () => {
      try {
        const res = await fetch(GAS_URL);
        appData = await res.json();
        initUI();
        document.getElementById('loader').style.display = 'none';
      } catch (e) { alert("โหลดข้อมูลไม่สำเร็จ: " + e); }
    };

    function initUI() {
      const schools = [...new Set(appData.students.map(s => s.school))].filter(Boolean).sort();
      const genders = [...new Set(appData.students.map(s => s.gender))].filter(Boolean).sort();
      const acts = appData.activities.map(a => a.กิจกรรม).filter(Boolean).sort();
      
      schools.forEach(s => document.getElementById('f-school').innerHTML += `<option value="${s}">${s}</option>`);
      genders.forEach(g => document.getElementById('f-gender').innerHTML += `<option value="${g}">${g}</option>`);
      acts.forEach(a => document.getElementById('f-activity').innerHTML += `<option value="${a}">${a}</option>`);
      
      renderStudents(appData.students);
      renderActivities();
      renderSchedule();
    }

    // ฟังก์ชันสำหรับเปิด/ปิด Hamburger Menu ในมือถือ
    function toggleMobileMenu() {
      document.getElementById('mobile-menu').classList.toggle('hidden');
    }

    // ปรับปรุงฟังก์ชันเปลี่ยนหน้า เพื่อให้จัดการสถานะปุ่มทั้ง Desktop และ Mobile
    function switchView(view) {
      // 1. เปลี่ยนหน้า View
      document.querySelectorAll('.view-section').forEach(s => s.classList.add('hidden'));
      document.getElementById(`view-${view}`).classList.remove('hidden');
      
      // 2. จัดการสีปุ่ม Desktop
      document.querySelectorAll('.desktop-nav').forEach(b => {
        b.classList.remove('nav-active', 'text-blue-600');
        b.classList.add('text-gray-500');
      });
      document.getElementById(`nav-${view}`).classList.add('nav-active', 'text-blue-600');
      document.getElementById(`nav-${view}`).classList.remove('text-gray-500');

      // 3. จัดการสีปุ่ม Mobile
      document.querySelectorAll('.mobile-nav').forEach(b => {
        b.classList.remove('text-blue-600', 'bg-blue-50');
        b.classList.add('text-gray-600');
      });
      document.getElementById(`mob-${view}` || `mob-nav-${view}`).classList.add('text-blue-600', 'bg-blue-50');
      document.getElementById(`mob-nav-${view}`).classList.remove('text-gray-600');

      // 4. ซ่อนเมนู Mobile เมื่อกดเลือกหน้าแล้ว
      document.getElementById('mobile-menu').classList.add('hidden');
    }

    function renderStudents(data) {
      const tbody = document.getElementById('student-table');
      tbody.innerHTML = data.map(s => `
        <tr class="hover:bg-blue-50 cursor-pointer transition whitespace-nowrap" onclick="openStudentDetail('${s.id}')">
          <td class="px-6 py-4 text-sm font-medium text-gray-900">${s.id}</td>
          <td class="px-6 py-4 text-sm text-gray-700">${s.firstName} ${s.lastName}</td>
          <td class="px-6 py-4 text-sm text-gray-600">${s.school}</td>
          <td class="px-6 py-4 text-sm text-gray-600">${s.gender}</td>
          <td class="px-6 py-4 flex flex-wrap gap-1">
            ${s.activities.length > 0 
              ? s.activities.map(a => `<span class="bg-blue-100 text-blue-700 text-[10px] px-2 py-0.5 rounded-full">${a}</span>`).join('')
              : '<span class="text-gray-400 text-xs">ไม่มีกิจกรรม</span>'}
          </td>
        </tr>
      `).join('');
    }

    function filterStudents() {
      const q = document.getElementById('search').value.toLowerCase();
      const sch = document.getElementById('f-school').value;
      const gen = document.getElementById('f-gender').value;
      const act = document.getElementById('f-activity').value;

      const filtered = appData.students.filter(s => {
        const mSearch = s.firstName.toLowerCase().includes(q) || s.lastName.toLowerCase().includes(q) || s.id.includes(q);
        const mSchool = !sch || s.school === sch;
        const mGender = !gen || s.gender === gen;
        const mAct = !act || s.activities.some(studentAct => String(studentAct) === String(act));
        
        return mSearch && mSchool && mGender && mAct;
      });
      renderStudents(filtered);
    }

    function renderActivities() {
      document.getElementById('activity-list').innerHTML = appData.activities.map(a => `
        <div class="bg-white p-6 rounded-xl shadow-sm border border-gray-100">
          <h3 class="text-xl font-bold text-blue-600 mb-2">${a.กิจกรรม}</h3>
          <p class="text-gray-600 text-sm mb-4">${a.รายละเอีอด}</p>
          <div class="text-xs text-gray-500 space-y-1 mb-4">
            <p><strong>วันที่:</strong> ${a.วันที่} | <strong>เวลา:</strong> ${a.เวลา}</p>
            <p><strong>สถานที่:</strong> ${a.สถานที่}</p>
          </div>
          <div class="map-container">${a['Link สำหรับสถานที่']}</div>
        </div>
      `).join('');
    }

    function renderSchedule() {
      document.getElementById('schedule-table').innerHTML = appData.schedule.map(s => `
        <tr class="whitespace-nowrap">
          <td class="px-6 py-4 text-sm text-gray-900 font-bold">${s['วันที่จัดกิจกรรม'] || s['days']}</td>
          <td class="px-6 py-4 text-sm text-gray-600">${s['เวลา'] || s['Time']}</td>
          <td class="px-6 py-4 text-sm text-blue-600 font-medium">${s['ชื่อกิจกรรม'] || s['activities']}</td>
          <td class="px-6 py-4 text-sm text-gray-500">${s['สถานที่'] || s['location']}</td>
        </tr>
      `).join('');
    }

    function openStudentDetail(id) {
      const s = appData.students.find(x => x.id === id);
      document.getElementById('m-student-name').innerText = `${s.firstName} ${s.lastName}`;
      document.getElementById('m-student-info').innerHTML = `รหัสนักเรียน: ${s.id} | สถาบัน: ${s.school} | เพศ: ${s.gender}`;
      
      const actDetails = s.activities.map(actName => {
        const detail = appData.activities.find(a => a.กิจกรรม === actName);
        return detail ? `
          <div class="bg-gray-50 p-4 rounded-lg border">
            <h4 class="font-bold text-blue-600">${detail.กิจกรรม}</h4>
            <p class="text-sm text-gray-600 mb-2">${detail.รายละเอีอด}</p>
            <p class="text-xs text-gray-500">📍 ${detail.สถานที่} (${detail.วันที่} ${detail.เวลา})</p>
          </div>
        ` : `<div class="bg-gray-50 p-4 rounded-lg border border-dashed"><h4 class="font-bold text-gray-500">${actName}</h4><p class="text-sm text-gray-400">ยังไม่มีรายละเอียดสำหรับกิจกรรมนี้</p></div>`;
      }).join('');

      document.getElementById('m-student-activities').innerHTML = actDetails || '<p class="text-gray-400 text-center py-4">ยังไม่ได้เข้าร่วมกิจกรรมใด</p>';
      document.getElementById('modal-student').classList.remove('hidden');
    }

    function closeModal() { document.getElementById('modal-student').classList.add('hidden'); }
  </script>
